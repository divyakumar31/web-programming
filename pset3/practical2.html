        <button type="submit">Submit</button>

    <?php
    if ($_SERVER['REQUEST_METHOD'] == "POST") {
        $number = $_REQUEST['number'];
        echo "<h1>Pattern 1</h1>";

        $char = 'a';
        $num = 1;

        for ($i = 1; $i <= $number; $i++) {
            for ($j = 1; $j <= $i; $j++) {
                if ($i % 2 != 0) {
                    echo $num++ . " ";
                } else {
                    echo $char++ . " ";
                }
            }
            echo "<br>";
        }

        echo "<h1>Pattern 2</h1>";

        for ($i = 1; $i <= $number; $i++) {
            for ($j = 1; $j <= $i; $j++) {
                echo ($i + $j) % 2 == 0 ? "1 " : "0 ";
            }
            echo "<br>";
        }
    }
    ?>
